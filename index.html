Q.No.2 Write code to explain prototype chaining

Output :
        const Address = {
            Country : function(){
                console.log("Hello I Live in India!!!")
            }
        }
        
        const SubAddress = Object.create(Address)
        SubAddress.State = function(){
                console.log("My City situated in Madhya Pradesh")
        }
        
        const LivingCity = Object.create(SubAddress)
        SubAddress.City = function(){
                console.log("I live in Bhopal City!!!!!!!")
        }
        
        const LivingHouse = Object.create(LivingCity)
        SubAddress.House = function(){
                console.log("My House No is A/2 215")
        }
        
        LivingHouse.City()

Q.No.3 Add a method to calculate sum of all elements in Array in array's protype, use that method to calculate sum for multiple arrays

Output :
            const Addition = {
                Sum : function(arr){
                        let sum = 0;
                        arr.forEach((item)=>{
                            sum += item
                        })
                        return sum
                }
            }
            
            Qno1 = Object.create(Addition)
            Qno2 = Object.create(Addition)
            
            console.log(Qno1.Sum([5,4,3]))
            console.log(Qno2.Sum([7,15,20,43]))
